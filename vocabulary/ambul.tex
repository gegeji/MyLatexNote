\chapter{ambul, ambl}
\begin{vocabulary}[ambulance]
\end{vocabulary}

\begin{vocabulary}[ambulate /ˈæm byəˌleɪt/] to walk about or move from place to place.
\end{vocabulary}

\begin{vocabulary}[ambulant /ˈæmbjələnt/] (of a patient) able to walk; not having to stay in bed
\end{vocabulary}

\begin{vocabulary}[preamble /priː'æmb(ə)l/]
\textit{Harding gave him the news without preamble} (=without saying anything else before it)
\end{vocabulary}
